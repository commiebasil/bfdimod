	desc = GPG_event.1.desc
	picture = GFX_GPG_event1
	
	is_triggered_only = yes
	
	option = {
		name = GPG_event.1.a
		add_political_power = 25
		hidden_effect = {
			if = {
				limit = { 
					is_ai = yes
					OR = {
						is_historical_focus_on = no	
						has_game_rule = { rule = GPG_ai_behavior option = RANDOM }
					} 
				}
				random_list = {
					40 = {
						set_country_flag = stability_save_path
					}
					30 = {
						set_country_flag = civil_war_path
					}
					30 = {
						set_country_flag = fries_path
					}
				}
			}
			else_if = {
				limit = {
					is_ai = yes
					OR = {
						has_game_rule = { rule = GPG_ai_behavior option = MAINTAIN_PEACE_NO_FRIES }
						has_game_rule = { rule = GPG_ai_behavior option = MAINTAIN_PEACE_FRIES }
					}		
				}
				set_country_flag = stability_save_path
			}
			else_if = {
				limit = {
					is_ai = yes
					OR = {
						has_game_rule = { rule = GPG_ai_behavior option = CIVIL_WAR_FRIES }
						has_game_rule = { rule = GPG_ai_behavior option = DEFAULT }			
					}
					is_historical_focus_on = yes	
				}
				set_country_flag = fries_path
			}
			else_if = {
				limit = {
					is_ai = yes
					OR = {
						has_game_rule = { rule = GPG_ai_behavior option = CIVIL_WAR_SOCIALIST }
						has_game_rule = { rule = GPG_ai_behavior option = CIVIL_WAR_ANARCHIST }	
						has_game_rule = { rule = GPG_ai_behavior option = CIVIL_WAR_FASCIST }	
						has_game_rule = { rule = GPG_ai_behavior option = CIVIL_WAR_YOYLELANDIA }
						has_game_rule = { rule = GPG_ai_behavior option = CIVIL_WAR_DEMOCRATIC }			
					}	
				}
				set_country_flag = civil_war_path
			}
			else = {
				set_country_flag = civil_war_path
			}
		}
	}

	option = {
		name = GPG_event.1.b
		add_political_power = 25
		country_event = GPG_event.2
		hidden_effect = {
			if = {
				limit = { is_ai = yes }
				random_list = {
					30 = {
						set_country_flag = stability_save_path
					}
					40 = {
						set_country_flag = civil_war_path
					}
					30 = {
						set_country_flag = fries_path
					}
				}
			}
		}

		ai_chance = {
			base = 0
		}
	}
}
