infantry_GPG = {

	available_for = {
		GPG
	}

	role = infantry

	upgrade_prio = {
		base = 5

		modifier = {
			factor = 1
		}
	}

	infantry_civil_war_GPG = {
		division_names_group = GPG_Inf_01
		upgrade_prio = {
			base = 1.5

			modifier = {
				factor = 1.5
			}
		}

		can_upgrade_in_field = {
            has_equipment = { artillery_equipment > 25 }
        }

		target_template = {	
			
			regiments = {
				infantry = 4
			}
		}

		replace_at_match = 0.75
        replace_with = infantry_civil_war_upgrade_GPG
        target_min_match = 0.75
	}

	infantry_civil_war_upgrade_GPG = {
		division_names_group = GPG_Inf_01
		upgrade_prio = {
			base = 1
			modifier = {
				factor = 0
				arms_factory < 3
			}

			modifier = {
				factor = 3
				arms_factory > 3
			}
		}

		can_upgrade_in_field = {
            has_equipment = { artillery_equipment > 25 }
			has_equipment = { support_equipment_1 > 25 }
        }

		target_template = {

			support = {
				artillery = 1
			}

			regiments = {
				infantry = 4
			}
		}

		replace_at_match = 0.75
        replace_with = infantry_early_GPG
        target_min_match = 0.75
	}

	infantry_early_GPG = {
		#Unlocked after civil war is done
		division_names_group = GPG_Inf_01
		upgrade_prio = {
			base = 2
			modifier = {
				factor = 3
			}
		}

		can_upgrade_in_field = {
            has_equipment = { artillery_equipment > 25 }
			has_equipment = { support_equipment_1 > 25 }
        }

		enable = {
			OR = {
				has_completed_focus = GPG_focus_never_again
				has_completed_focus = GPG_focus_third_congress
			}
		}

		target_template = {

			support = {
				engineer = 1
				artillery = 1
			}

			regiments = {
				infantry = 5
			}
		}

		replace_at_match = 0.75
        replace_with = infantry_early_upgrade_GPG
        target_min_match = 0.75
	}

	infantry_early_upgrade_GPG = {
		division_names_group = GPG_Inf_01
		upgrade_prio = {
			base = 3
			modifier = {
				factor = 0
				arms_factory < 7
			}

			modifier = {
				factor = 3
				arms_factory > 7
			}
		}

		enable = {
			OR = {
				has_completed_focus = GPG_focus_never_again
				has_completed_focus = GPG_focus_third_congress
			}
		}

		target_template = {

			support = {
				engineer = 1
				logistics_company = 1
				field_hospital = 1
				artillery = 1
			}

			regiments = {
				infantry = 5
			}
		}
	}
}
