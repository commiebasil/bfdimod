			}
		}

		replace_at_match = 0.75

	}

		#Unlocked after civil war is done

			base = 2

			has_equipment = { support_equipment_1 > 25 }
        }

		enable = {
			NOT = { country_exists = GPG }
			NOT = { country_exists = GSR }
			NOT = { country_exists = GDR }
		}

		target_template = {
			
			support = {
				artillery = 1
			}

			regiments = {
				infantry = 5
			}
		}

		replace_at_match = 0.75
        replace_with = infantry_early_upgrade_GFG
        target_min_match = 0.75
	}

	infantry_early_upgrade_GFG = {
		division_names_group = GFG_Inf_01
		upgrade_prio = {
			base = 3
			modifier = {
				factor = 0
				arms_factory < 8
			}

			modifier = {
				factor = 3
				arms_factory > 8
			}
		}


		target_template = {

			support = {
				artillery = 1
				engineer = 1
			}

			regiments = {
				infantry = 5
			}
