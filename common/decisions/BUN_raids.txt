
decision_BUN_raids_category = {
	BUN_raid_state_85 = {
		icon = GFX_decision_generic_confiscation
		state_target = yes
        targets = { state = 202 210 214 }
        on_map_mode = map_only
        
		visible = {
			controls_state = 85
			NOT = {controls_state = FROM}
			NOT = { controller = {is_guaranteed_by = ROOT}}
			NOT = { has_military_access_to = controller}
			NOT = {is_puppet = yes}
			has_border_war = no
			has_war = no
			NOT = {has_cosmetic_tag = BUN_revolt}
		}

		fire_only_once = no
		cost = 70
		days_re_enable = 200

		complete_effect = {
			FROM = {
				damage_building = {
 					type = infrastructure
  					damage = 0.5
				}
				hidden_effect = {
					IF = {
						limit = {controller = {is_ai = yes}}						
						random = {
							chance = 15
							controller = {
								diplomatic_relation = {
									country = BUN
									relation = embargo
									active = yes
								}
							}
						}
					}
				}
			}
			
			if = {
				limit = {
					is_ai = yes
				}
				set_variable = { var_currently_raiding = 1 }
				country_event = {
					id = BUN_borderwars.1
					days = 40
					random_hours = 48
				}
			}

			IF = {
				limit = {
					YFC = {has_completed_focus = YFC_focus_demand_end_pillage}
					YFC = {controls_state = FROM}
				}
				start_border_war = {
					change_state_after_war = no
					attacker = {
						state = 85
						num_provinces = 1
						on_win = YFC_borderwars.1
						on_lose = YFC_borderwars.2
						modifier = 0.25
						terrain_factor = 0.5			
					}	
					defender = {
						state = FROM
						num_provinces = 1
						on_win = YFC_borderwars.3
						on_lose = BUN_borderwars.5
						dig_in_factor = 2
					}
				}
			}
			ELSE = {
				start_border_war = {
					change_state_after_war = no
					attacker = {
						state = 85
						num_provinces = 1
						on_win = YFC_borderwars.1
						on_lose = YFC_borderwars.2
						modifier = 0.65	
						terrain_factor = 0.5			
					}	
					defender = {
						state = FROM
						num_provinces = 1
						on_win = YFC_borderwars.3
						on_lose = BUN_borderwars.5
						dig_in_factor = 2
					}
				}
			}
		}

		ai_will_do = {
			base = 0
			modifier = {
				add = 15
				has_political_power > 70
			}
			modifier = {
				add = -10
				AND = {
					FROM = {controller = {is_ai = no}}
					FROM = {controller = {has_war = yes}}
				}
			}
		}
	}
