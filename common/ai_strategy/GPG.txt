## CIVIL WAR ##

GPG_civil_war = {
    #Fun fact! Having more divs are better then having no divs
	allowed = {
		original_tag = GPG
	}
	enable = {
		country_exists = GFG
		country_exists = GDR 
		country_exists = GSR
	}

	abort = {
		has_war = no
	}

	ai_strategy = {
		type = production_upgrade_desire_offset
		id = infantry_equipment_1
		value = -1000
	}
	ai_strategy = {
		type = equipment_production_factor
		id = infantry
		value = 65
	}
	ai_strategy = {
		type = ignore_army_incompetence
		value = 50
	}
	ai_strategy = {
		type = force_build_armies
		value = 20
	}
}

GPG_GPR_trade = {
	allowed = {
		original_tag = GPG
	}
	enable = {
		has_completed_focus = GPG_focus_cabinet_resisted
		has_idea = idea_GPG_factionalism
		has_war = no
	}

	abort = {
		OR = {
			has_war = yes
			NOT = {has_idea = idea_GPG_factionalism}
		}	
	}

	ai_strategy = {
		type = put_unit_buffers
		ratio = 1000
		states = { 145 }
		subtract_fronts_from_need = yes
	}
}

		original_tag = GPG

			has_completed_focus = GPG_focus_never_again

		has_army_size = {
            size > 15
        }
	}

		value = 250
