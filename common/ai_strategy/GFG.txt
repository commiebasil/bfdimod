## CIVIL WAR ##

GFG_civil_war = {
    #Fun fact! Having more divs are better then having no divs
	allowed = {
		original_tag = GFG
	}
	enable = {
		country_exists = GPG
		country_exists = GDR 
		country_exists = GSR
	}

	abort = {
		has_war = no
	}

	ai_strategy = {
		type = production_upgrade_desire_offset
		id = infantry_equipment_1
		value = -1000
	}
	ai_strategy = {
		type = equipment_production_factor
		id = infantry
		value = 65
	}
	ai_strategy = {
		type = ignore_army_incompetence
		value = 50
	}
	ai_strategy = {
		type = force_build_armies
		value = 30
	}
}

## ARMY PLACEMENT ##

GFG_rush_GPR = {
    #GFG will rush war on GPR 
	allowed = {
		original_tag = GFG
	}
	enable = {
		focus_progress = {
			focus = GFG_focus_break_the_line
			progress > 0.01
		}
	}

	abort = {
		has_war = yes
	}

	ai_strategy = {
		type = prepare_for_war
		id = GPR
		value = 300
	}
}
